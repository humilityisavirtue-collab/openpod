Quick Start

Routing
from openpod import KlawRouter

router = KlawRouter()
result = router.route("How do I center a div?")

print(result["response"])   # The answer
print(result["cost"])       # $0.0001
print(result["savings"])    # $0.0029 saved
print(result["tier_name"])  # "template" (free)
Agent Comms
from openpod import Pod, Agent

# Pod-based messaging
pod = Pod("my_cell", roles=["alpha", "beta"])
pod.send("alpha", "beta", "Build it", suit="+7D")
msgs = pod.inbox("beta")

# Full agent with K-routed thinking
agent = Agent("my-agent")

@agent.on_message
def handle(msg):
    return agent.think(msg.text)
